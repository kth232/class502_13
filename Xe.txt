--기존에 사용했던 테이블 삭제
DROP TABLE MEMBER;
DROP SEQUENCE SEQ_MEMBER;

CREATE TABLE MEMBER (
	USER_NO NUMBER(10) PRIMARY KEY,
	USER_ID VARCHAR2(30) UNIQUE NOT NULL,
	USER_PW VARCHAR2(65) NOT NULL,
	USER_NM VARCHAR2(40) NOT NULL,
	MOBILE VARCHAR2(15),
	REG_DT DATE DEFAULT SYSDATE, --기본값으로 SYSDATE
	MOD_DT DATE 
);

CREATE SEQUENCE SEQ_MEMBER;

SELECT * FROM MEMBER;

CREATE OR REPLACE PROCEDURE REGISTER_MEMBER(
	P_USER_ID VARCHAR2,
	P_USER_PW VARCHAR2,
	P_USER_NM VARCHAR2,
	P_MOBILE VARCHAR2
)
IS 
BEGIN 
	INSERT INTO MEMBER (USER_NO, USER_ID, USER_PW, USER_NM, MOBILE)
		VALUES (SEQ_MEMBER.NEXTVAL, 
		P_USER_ID, P_USER_PW, P_USER_NM, P_MOBILE);
	
END REGISTER_MEMBER;
